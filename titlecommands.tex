\newcommand\textunderset[2]
{
	\leavevmode
	\vtop{\offinterlineskip
		\halign{%
			\hfil##\hfil\cr
			\strut#2\cr
			\noalign{\kern.5ex}
			\dynscriptsize\strut#1\cr
		}%
	}%
}
